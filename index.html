Лог событий

События Tilda

Tilda генерирует кастомные события через document. Подписывайтесь на них через addEventListener.

tildaready

Срабатывает когда Tilda полностью загружена и все блоки инициализированы.

document.addEventListener('tildaready', function() {
  console.log('Tilda готова!');
});

allanimationend

Срабатывает когда заканчиваются все анимации на странице.

document.addEventListener('allanimationend', function() {
  console.log('Все анимации завершены');
});

scrollto

Позволяет программно скроллить к блоку по его ID.

// Скроллить к блоку rec123456
var event = new CustomEvent('scrollto', {
  detail: { blockid: 'rec123456' }
});
document.dispatchEvent(event);

tildaformsent

Срабатывает после успешной отправки любой формы на странице.

document.addEventListener('tildaformsent', function(e) {
  console.log('Форма отправлена:', e.detail);
});

popupshown / popuphidden

Срабатывают при открытии и закрытии попапов.

document.addEventListener('popupshown', function(e) {
  console.log('Попап открылся:', e.detail.popupid);
});
document.addEventListener('popuphidden', function(e) {
  console.log('Попап закрылся:', e.detail.popupid);
});

productaddedtocart / productremovedfromcart

События корзины: добавление и удаление товара.

document.addEventListener('productaddedtocart', function(e) {
  var product = e.detail;
  console.log('Добавлен:', product.title, product.price);
});

Работа с формами

Tilda предоставляет API для управления формами, добавления полей и перехвата данных.

Перехват перед отправкой

Используйте window.tildaFormsCallback для перехвата данных формы перед отправкой.

window.tildaFormsCallback = function(formData) {
  // formData — объект с данными формы
  console.log('Данные формы:', formData);
  // Можно модифицировать данные
  formData['custom_field'] = 'значение';
  // Вернуть true чтобы отправить, false — отменить
  return true;
};

Демо-форма

Симуляция формы Tilda с живым перехватом данных.

После отправки формы

Подписка на событие успешной отправки для аналитики или редиректа.

document.addEventListener('tildaformsent', function(e) {
  var detail = e.detail;
  // detail.formid — ID формы
  // detail.data  — данные формы

  // Отправка в GTM
  window.dataLayer = window.dataLayer || [];
  window.dataLayer.push({
    event: 'form_submit',
    form_id: detail.formid
  });

  // Редирект
  window.location.href = '/thank-you';
});

Динамическое добавление UTM-меток

Добавляем скрытые поля с UTM-метками в форму.

document.addEventListener('tildaready', function() {
  var utmFields = ['utm_source', 'utm_medium',
                   'utm_campaign', 'utm_term',
                   'utm_content'];
  var params = new URLSearchParams(location.search);

  document.querySelectorAll('.t-form').forEach(function(form) {
    utmFields.forEach(function(field) {
      var val = params.get(field);
      if (val) {
        var input = document.createElement('input');
        input.type = 'hidden';
        input.name = field;
        input.value = val;
        form.appendChild(input);
      }
    });
  });
});

Управление попапами

Открывайте и закрывайте попапы программно через глобальные функции Tilda.

Открыть попап

Функция t_popup_show — открывает попап по его ID.

// Открыть попап с ID блока rec123456
t_popup_show('rec123456');

// Или через событие
var event = new CustomEvent('showpopup', {
  detail: { popupid: 'rec123456' }
});
document.dispatchEvent(event);

Закрыть попап

Функция t_popup_hide — закрывает попап.

// Закрыть попап
t_popup_hide('rec123456');

// Закрыть все попапы
document.querySelectorAll('.t-popup').forEach(function(popup) {
  t_popup_hide(popup.getAttribute('id'));
});

Попап по условию

Открываем попап после задержки или по скроллу.

// Попап через 5 секунд
setTimeout(function() {
  t_popup_show('rec123456');
}, 5000);

// Попап при скролле 50%
window.addEventListener('scroll', function() {
  var scrolled = window.scrollY;
  var total = document.body.scrollHeight - window.innerHeight;
  if (scrolled / total > 0.5) {
    t_popup_show('rec123456');
    this.removeEventListener('scroll', arguments.callee);
  }
});

Следить за корзиной

document.addEventListener('productaddedtocart', function(e) {
  updateCartCounter();
});
document.addEventListener('productremovedfromcart', function(e) {
  updateCartCounter();
});

function updateCartCounter() {
  var cart = JSON.parse(
    localStorage.getItem('tcart') || '[]'
  );
  var count = cart.reduce(function(n, i) {
    return n + i.quantity;
  }, 0);
  document.querySelector('.cart-count').textContent = count;
}

Zero Block (Редактор блоков)

Zero Block позволяет писать HTML/CSS/JS прямо в блоке. Вот паттерны для работы с ним.

Структура Zero Block


<div id="rec123456" class="r t-rec">
  <div class="t-container">
    <div class="t-col t-col_12">
      <div class="t-text">
        <!-- Ваш HTML -->
      </div>
    </div>
  </div>
</div>

JS в Zero Block выполняется после загрузки блока.

Инициализация в Zero Block

Правильный способ инициализации скриптов в Zero Block.

// Безопасный способ — дождаться готовности DOM
document.addEventListener('DOMContentLoaded', function() {
  initMyBlock();
});

// Если Tilda уже загружена
document.addEventListener('tildaready', function() {
  initMyBlock();
});

function initMyBlock() {
  var block = document.querySelector('#rec123456');
  if (!block) return;
  // Ваш код...
}

Анимации и эффекты

Управление анимацией блоков

// Tilda использует классы для анимаций
// t-animate — элемент анимируется
// t-animate_started — анимация запущена

// Принудительно запустить анимацию
document.querySelectorAll('.t-animate').forEach(function(el) {
  el.classList.add('t-animate_started');
});

// Сбросить и перезапустить
el.classList.remove('t-animate_started');
setTimeout(function() {
  el.classList.add('t-animate_started');
}, 100);

ScrollReveal паттерн

Анимация элементов при скролле (как в Tilda).

document.addEventListener('tildaready', function() {
  var observer = new IntersectionObserver(function(entries) {
    entries.forEach(function(entry) {
      if (entry.isIntersecting) {
        entry.target.classList.add('is-visible');
        observer.unobserve(entry.target);
      }
    });
  }, { threshold: 0.2 });

  document.querySelectorAll('.animate-on-scroll')
    .forEach(function(el) {
      observer.observe(el);
    });
});

Cookie и localStorage

// Tilda сохраняет данные пользователя в localStorage
// Ключи которые использует Tilda:
// 'tcart'       — данные корзины
// 'tilda_utm'   — UTM метки
// 'tws'         — данные авторизации

// Установить cookie
function setCookie(name, value, days) {
  var expires = '';
  if (days) {
    var date = new Date();
    date.setTime(date.getTime() + days*24*60*60*1000);
    expires = '; expires=' + date.toUTCString();
  }
  document.cookie = name + '=' + value + expires + '; path=/';
}

// Получить cookie
function getCookie(name) {
  var nameEQ = name + '=';
  return document.cookie.split(';')
    .map(function(c) { return c.trim(); })
    .find(function(c) { return c.indexOf(nameEQ) === 0; })
    ?.substring(nameEQ.length) || null;
}

Мета-данные блоков

Каждый блок Tilda имеет data-атрибуты с полезной информацией.

// Получить все блоки страницы
document.querySelectorAll('.r.t-rec').forEach(function(block) {
  var id = block.id;           // rec123456
  var type = block.dataset.record_type;  // тип блока
  var bg = block.dataset.bgcolor;        // цвет фона

  console.log(id, type, bg);
});

// Найти блок по ID
var block = document.getElementById('rec123456');
var blockData = {
  id: block.id,
  type: block.getAttribute('data-record-type'),
  animationType: block.getAttribute('data-animationtype')
};
